# CMakeLists.txt 파일 내용

# CMake 최소 버전 설정
cmake_minimum_required(VERSION 3.10)

# 프로젝트 이름 설정
project(sandbox)

# 실행 파일 추가
add_executable(sandbox main.c util.c)

# 테스트 타겟 추가
add_executable(tests test_util.c util.c)
target_link_libraries(tests gtest)

# 테스트 타겟 실행
enable_testing()
add_test(NAME UtilTest COMMAND tests)
